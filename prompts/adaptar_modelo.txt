# Campos de entrada
col1, col2, col3 = st.columns(3)

with col1:
    area = st.number_input("Área (ha)", min_value=100, max_value=5000000, value=10000)
    tipo_solo = st.selectbox("Tipo de Solo", ["argiloso", "arenoso", "misto"])

with col2:
    precipitacao = st.slider("Precipitação (mm)", 800, 2000, 1400)
    temperatura = st.slider("Temperatura (°C)", 18, 32, 25)

with col3:
    altitude = st.number_input("Altitude (m)", 200, 1200, 500)

# Pré-processamento
mapa_solo = {"argiloso": 0, "arenoso": 1, "misto": 2}
solo_encoded = mapa_solo[tipo_solo]

# Previsão
if st.button("Calcular Previsão"):
